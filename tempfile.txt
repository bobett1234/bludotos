  SimpleDock.prototype.findmenuClick = function (onclicked, addapp){
  for (var d=0; d < imageDetails[addapp].menuClick.length d++) {
  if (imageDetails[addapp].menuClick[d] == onclicked) {
  return d;
  } else {
  return false;
  }
  }
  }
  SimpleDock.prototype.findmenuItems = function (name, addapp){
  for (var d=0; d < imageDetails[addapp].menuItems.length d++) {
  if (imageDetails[addapp].menuItems[d] == name) {
  return d;
  } else {
  return false;
  }
  }
  }
  SimpleDock.prototype.removeclick = function (app, name, onclicked){
  var addapp = this.findApp(app);
  var aname = this.findmenuItems(name[0], addapp);
  var aonclicked = this.findmenuClick(onclicked[0], addapp);
  imageDetails[addapp].menuItems.splice(aname, 1);
  imageDetails[addapp].menuClick.splice(aonclicked, 1);
  alert(imageDetails[addapp].menuItems);
  alert(imageDetails[addapp].menuClick);
  }
